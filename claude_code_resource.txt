11c. camera_screen.dart — Template

import 'package:camera/camera.dart';

import 'package:flutter/material.dart';

import 'package:flutter_screenutil/flutter_screenutil.dart';

import 'app_colors.dart';



class CameraScreen extends StatefulWidget {

  final Function(String imagePath) onPhotoCaptured;

  const CameraScreen({required this.onPhotoCaptured, super.key});

  @override State<CameraScreen> createState() => _CameraScreenState();

}



class _CameraScreenState extends State<CameraScreen> {

  CameraController? _controller;

  bool _flashOn = false;



  @override void initState() {

    super.initState();

    _initCamera();

  }



  Future<void> _initCamera() async {

    final cameras = await availableCameras();

    _controller = CameraController(cameras.first, ResolutionPreset.high);

    await _controller!.initialize();

    if (mounted) setState(() {});

  }



  Future<void> _capturePhoto() async {

    if (_controller == null || !_controller!.value.isInitialized) return;

    final file = await _controller!.takePicture();

    widget.onPhotoCaptured(file.path);

  }



  @override Widget build(BuildContext context) {

    return Scaffold(

      backgroundColor: Colors.black,

      body: Stack(children: [

        if (_controller?.value.isInitialized == true)

          Positioned.fill(child: CameraPreview(_controller!)),

        // Flash toggle — x=320, y=40

        Positioned(top: 40.h, right: 16.w,

          child: IconButton(

            icon: Icon(_flashOn ? Icons.flash_on : Icons.flash_off,

              color: Colors.white, size: 24.sp),

            onPressed: () => setState(() { _flashOn = !_flashOn; }),

          )),

        // Shutter — x=148, y=616, 64×64dp

        Positioned(bottom: 40.h, left: 148.w,

          child: GestureDetector(

            onTap: _capturePhoto,

            child: Container(width: 64.w, height: 64.h,

              decoration: BoxDecoration(shape: BoxShape.circle,

                color: AppColors.ctaBg)),

          )),

      ]),

    );

  }

}

14. CRITICAL NOTES FOR CLAUDE CODE



Hindi / Devanagari Script

Your designs contain Hindi text: आधार कार्ड को इस क्षेत्र में रखें, कस्टमर प्रतीक्षा कर रहे हैं, etc. Noto Sans is the ONLY acceptable font — it provides full Devanagari coverage. Never substitute Roboto or any other font. Ensure all string literals containing Hindi are UTF-8 encoded Dart strings.



Elevation & Shadow

Two shadow levels are defined in the design system:

1dp shadow: BoxShadow(color: Color(0x26000000), offset: Offset(0, 1), blurRadius: 3)

2dp shadow: BoxShadow(color: Color(0x26000000), offset: Offset(0, 2), blurRadius: 6)



ScreenUtil Initialization

Initialize ScreenUtilInit at app root with designSize: Size(360, 720). Use .w for horizontal, .h for vertical, .sp for font sizes, .r for border radius. Never use hardcoded pixel values.



Figma Node Fetching Strategy
